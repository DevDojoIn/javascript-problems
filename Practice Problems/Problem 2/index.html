Problem 2:Write a js program to display area and perimeter of a rectangle.

Solution:


      function calculateRectangle(length, width) {
        // Calculate area and perimeter
        const area = length * width;
        const perimeter = 2 * (length + width);
      
        // Display the results
        console.log("Area: " + area + " square units");
        console.log("Perimeter: " + perimeter + " units");
      }
      
      // Example usage
      calculateRectangle(10, 5);
